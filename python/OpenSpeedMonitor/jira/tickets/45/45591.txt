*Summary of Excel import/export cURL commands*

iteraplan version 3:

{noformat}#Login
curl -k -v --data ""j_username=USERNAME&j_password=PASSWORD"" PROTOCOL://HOST:PORT/iteraplan/j_iteraplan_security_check --cookie-jar cookies.txt

#Upload / Import
curl -X POST -k -v --cookie cookies.txt --data-binary @iteraplanExcelData.xls PROTOCOL://HOST:PORT/iteraplan/rest/excelImport

#Download / Export
curl -X POST -k -L -v --cookie cookies.txt -d ""_eventId=downloadAllData"" -o downloaded_model.xls PROTOCOL://HOST:PORT/iteraplan/show/import
{noformat}


iteraplan version 4:

{noformat}#Login
curl -k -v --data ""j_username=system&j_password=password"" PROTOCOL://HOST:PORT/CONTEXT/j_iteraplan_security_check --cookie-jar cookies.txt

#Upload / Import
curl -X POST -k -v --cookie cookies.txt -H ""Content-Type: application/vnd.ms-excel"" --data-binary @model.xls PROTOCOL://HOST:PORT/CONTEXT/data/model

#Download / Export
curl -k -L -v --cookie cookies.txt -o downloaded_model.xls PROTOCOL://HOST:PORT/CONTEXT/data/model.xls
{noformat}"
