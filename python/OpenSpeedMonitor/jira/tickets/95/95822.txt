{code:java}

<repository>

   <id>snapshots</id>

   <name>Nexus Snapshot Repository</name>

   <url>http://iteraplan.iteratec.de/nexus/repository/maven-snapshots/</url>

   <releases>

      <enabled>false</enabled>

   </releases>

   <snapshots>

      <enabled>true</enabled>

      <updatePolicy>always</updatePolicy>

   </snapshots>

</repository>

{code}
